107 public Selectable GetSelectable(Vector3 direction)
108 {

110
111 if (result == null || transform.parent == null || !IsActive())

113 return result;
114 }
115
116 var hexLayout = transform.parent.GetComponent<HexLayoutGroup>();
117 var directionAxis = GetAxis(direction);
118
119 if (hexLayout == null || directionAxis != (int)hexLayout.CellOrientation)
120 {
121 return result;
122 }
123
124 SetSiblingSelectables();
125 var startAxis = (int)hexLayout.StartAxis;
126 var rows = hexLayout.RowCount(SiblingSelectables.Count);
127 var columns = hexLayout.ColumnCount(SiblingSelectables.Count);
128 var (row, column) = GetIndex(rows, columns, startAxis);

130

132 {
133 return result;
134 }

147 else

149 }
150
